
FOR_STATMENT
    init code
    jmp for_begin
    loop_begin
    check_data
        true -> for_begin
        false -> loop_end
    for_suf
    suf data
    jmp loop_begin
    for_begin
    code
    jmp for_suf
    loop_end

WHILE_STATMENT
    loop_begin
    check_data
        false -> loop_end
    code
    jmp loop_begin
    loop_end

IF_STATMENT
    check_data
        false->if_end
    if_begin
    code
    jmp else_end
    if_end 
    else_code
    else_end 

SWITCH 
    exp (stack_top)

    push c
    pop i pop c 
    bneq i,c, case_end
    push i
    label case_end 

    push c
    pop i pop c 
    bneq i,c, case_end
    push i
    label case_end 

    break 
    jmp switch_end 

    label switch_end 
    pop i

label 
jmp 

DECL_LIST
    push type
    
    push id 
    push val
    pop val 
    pop id 
    pop type 
    decl id type 
    assign id = val 
    push type

    push id 
    push int_const
    pop int_const
    pop type  
    decl id type int_const
    
    initlist begin
    
    push exp
    pop exp
    add initlist val

    initlist end
